Basic Syntax

Hello World Program

#include 
int main() {
    printf("Hello, World!\n");
    return 0;
}

Output: Hello, World!

  • #include : Imports the standard input/output library.
  • int main(): Entry point of the program.
  • printf(): Outputs text to the console.
  • return 0: Signals successful program completion.

Variables

Declaration:

data_type variable_name;
data_type variable_name = value;

Variable Types:

  • Local: Defined within a function or block.
  • Global: Defined outside functions, accessible throughout the program.
  • Static: Retains value between function calls.
  • Extern: Declares a variable defined elsewhere.
  • Register: Suggests storage in CPU registers for faster access.

Naming Rules:

  • Use alphabets, digits, or underscores.
  • Start with a letter or underscore.
  • Avoid reserved keywords and whitespace.

Input/Output

Output: printf()

printf("format_string", arguments);

Input: scanf()

scanf("format_string", &variable);

Example:

#include 
int main() {
    int id;
    char name[50];
    printf("Enter ID: ");
    scanf("%d", &id);
    printf("Enter Name: ");
    scanf("%s", name);
    printf("ID: %d, Name: %s\n", id, name);
    return 0;
}

Output (Example Input): ID: 101, Name: Alice

Conditional Statements

if, if-else, switch

if (condition) {
    // code
} else {
    // code
}

switch (expression) {
    case value1:
        // code
        break;
    default:
        // code
}

Example:

#include 
int main() {
    int x = 5;
    if (x > 0)
        printf("Positive\n");
    else
        printf("Non-positive\n");
    return 0;
}

Output: Positive

Loops

for, while, do-while

for (initialization; condition; update) {
    // code
}

while (condition) {
    // code
}

do {
    // code
} while (condition);

Example:

#include 
int main() {
    for (int i = 1; i <= 5; i++) {
        printf("%d ", i);
    }
    return 0;
}

Output: 1 2 3 4 5

Arrays

data_type array_name[size];

Example:

int main() {
    int arr[5] = {10, 20, 30, 40, 50};

Dynamic Memory Management

FunctionPurposeSyntax
malloc()Allocates memorymalloc(size_t size);
calloc()Allocates and zeros memorycalloc(size_t num, size_t size);
realloc()Resizes memoryrealloc(void *ptr, size_t size);
free()Frees memoryfree(ptr);

Example:

#include 
#include 
int main() {
    int *arr = (int *)malloc(5 * sizeof(int));
    for (int i = 0; i < 5; i++) arr[i] = i + 1;
    printf("Array: ");
    for (int i = 0; i < 5; i++) printf("%d ", arr[i]);
    free(arr);
    return 0;
}

Output: Array: 1 2 3 4 5

Structures

struct struct_name {
    data_type member1;
    data_type member2;
};

Example:

#include 
struct Student {
    int id;
    char name[50];
    float score;
};
int main() {
    struct Student s = {101, "Alice", 95.5};
    printf("ID: %d, Name: %s, Score: %.1f\n", s.id, s.name, s.score);
    return 0;
}

Output: ID: 101, Name: Alice, Score: 95.5

Unions

union union_name {
    data_type member1;
    data_type member2;
};

Example:

#include 
union Data {
    int i;
    float f;
};
int main() {
    union Data d;
    d.i = 42;
    printf("Integer: %d\n", d.i);
    d.f = 3.14;
    printf("Float: %.2f\n", d.f);
    return 0;
}

Output:
Integer: 42
Float: 3.14

Enumerations

enum enum_name {value1, value2, value3};

Example:

#include 
enum Day {MON, TUE, WED, THU, FRI};
int main() {
    enum Day today = WED;
    printf("Day: %d\n", today);
    return 0;
}

Output: Day: 2

File Handling

  • Open/Create: fopen("file.txt", "mode")
  • Read: fscanf(), fgets()
  • Write: fprintf(), fputs()
  • Close: fclose()

Example:

#include 
int main() {
    FILE *fp = fopen("example.txt", "w");
    fprintf(fp, "Hello, File!\n");
    fclose(fp);
    return 0;
}
